    params.callback = cbAd;

    var url = API + '?' + Object.keys(params).map(function(k){
      return encodeURIComponent(k) + '=' + encodeURIComponent(
        typeof params[k] === 'object' ? JSON.stringify(params[k]) : String(params[k])
      );
    }).join('&');

    window[cbAd] = function(data) {
      delete window[cbAd];
      document.head.removeChild(script);
      callback(null, data);
    };

    var script = document.createElement('script');
    script.src = url;
    script.onerror = function() {
      delete window[cbAd];
      try { document.head.removeChild(script); } catch(e){}
      callback(new Error('Bağlantı hatası'));
    };
    document.head.appendChild(script);
  }

      sel.appendChild(f);

      dur.className='odurum ok'; dur.textContent='✓ '+s.okullar.length+' okul yüklendi';
    });
  });
